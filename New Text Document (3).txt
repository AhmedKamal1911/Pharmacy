model Medicine {
  id          String   @id @default(uuid())
  name        String
  code        String?  @unique

// --- إضافات المخزون ---
  location        String?  // مكان الرف (مثال: A1-Shelf-3)
  minStock        Int      @default(0) // الحد الأدنى
  maxStockLevel   Int?     // الحد الأقصى (اختياري)
  reorderPoint    Int?     // نقطة إعادة الطلب (غالباً بتكون أعلى من الحد الأدنى بشوية)
  
  // ربط الدواء بالقسم
  categoryId  String
  // استخدام Restrict يمنع الحذف من قاعدة البيانات لو القسم جواه أدوية
  category    Category @relation(fields: [categoryId], references: [id], onDelete: Restrict)

  description String?
  unit        String?
  minStock    Int?     @default(0)

  variants    MedicineVariant[] @relation("MedicineVariants")
  defaultVariantId String?          @unique
  defaultVariant   MedicineVariant? @relation("DefaultMedicineVariant", fields: [defaultVariantId], references: [id])

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model MedicineVariant {
  id          String    @id @default(uuid())
  
  medicineId  String
  medicine    Medicine  @relation("MedicineVariants", fields: [medicineId], references: [id], onDelete: Cascade)
  
  price       Decimal   @db.Decimal(10, 2) // السعر
  cost        Decimal   @db.Decimal(10, 2) // التكلفة
  stock       Int       @default(0)
  expiryDate  DateTime?
  batchNumber String?

  // العلاقة العكسية لمعرفة إذا كان هذا الـ Variant هو الافتراضي لدواء معين
  defaultForMedicine Medicine? @relation("DefaultMedicineVariant")

  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@unique([medicineId, batchNumber])
}

type CustomerType = "PERSON" | "COMPANY";

model Customer {
  id                String       @id @default(uuid())
  name              String
  
  // رقم التليفون غالباً بيكون Unique عشان متسجلش العميل مرتين

  // علاقة 1-to-1 مع نظام الولاء الخاص بالعميل
  loyalty           Loyalty?

  createdAt         DateTime     @default(now())
  updatedAt         DateTime     @updatedAt
}

  PARTIALLY_RETURNED

// --- فاتورة المشتريات الرئيسية ---
model PurchaseInvoice {
  id               String         @id @default(uuid())
  serialNumber     Int            @default(autoincrement()) @unique
  invoiceNumber    String         @unique 
  invoiceDate      DateTime
  
  // بيانات المورد (مع Snapshot لاسمه)
  supplierId       String
  supplier         Supplier       @relation(fields: [supplierId], references: [id], onDelete: Restrict)
  supplierName     String         

  status           PurchaseStatus @default(PAID)
  notes            String?

  // --- Invoice Totals (تم تفريغها هنا) ---
  itemsValue       Decimal        @db.Decimal(10, 2)
  profitPercentage Decimal        @db.Decimal(5, 2)
  total            Decimal        @db.Decimal(10, 2)
  baseTotal        Decimal        @db.Decimal(10, 2)
  taxTotal         Decimal        @db.Decimal(10, 2)
  extraCosts       Decimal        @default(0) @db.Decimal(10, 2)
  extraDiscount    Decimal        @default(0) @db.Decimal(10, 2)

  // الأصناف التابعة للفاتورة
  items            PurchaseItem[]

  createdAt        DateTime       @default(now())
  updatedAt        DateTime       @updatedAt
}

// --- أصناف الفاتورة (الـ History / Snapshot) ---
model PurchaseItem {
  id                 String           @id @default(uuid())
  
  // الفاتورة التابع لها
  purchaseInvoiceId  String
  purchaseInvoice    PurchaseInvoice  @relation(fields: [purchaseInvoiceId], references: [id], onDelete: Cascade)

  // الدواء الأساسي (مع Snapshot)
  medicineId         String
  medicine           Medicine         @relation(fields: [medicineId], references: [id], onDelete: Restrict)
  medicineName       String           
  medicineCode       String?          

  // المتغير (Variant) (مع Snapshot)
  variantId          String?
  variant            MedicineVariant? @relation(fields: [variantId], references: [id], onDelete: SetNull)
  variantPrice       Decimal?         @db.Decimal(10, 2) 
  variantExpiryDate  DateTime?        
  variantBatchNumber String?          

  // الكميات
  quantity           Int
  unitsPerPackage    Int
  bonus              Int              @default(0)

  // الحسابات الخاصة بالصنف في هذه الفاتورة تحديداً
  salePrice          Decimal          @db.Decimal(10, 2)
  tax                Decimal          @db.Decimal(10, 2)
  mainDiscount       Decimal          @db.Decimal(10, 2)
  extraDiscount      Decimal          @db.Decimal(10, 2)
  cost               Decimal          @db.Decimal(10, 2)

  expiryDate         DateTime?
  expirable          Boolean          @default(true)

  createdAt          DateTime         @default(now())
  updatedAt          DateTime         @updatedAt
}

  COMPLETED
  PENDING

}

  CHECK
  BANK_TRANSFER
  CREDIT
}

enum PaymentStatus {
  PAID
  PENDING

}

// 1. المورد (الكيان الأساسي)
model Supplier {
  id                  String              @id @default(uuid())
  short               String
  name                String
  supplierType        SupplierType
  
  // المديونية الحالية (تُحدث تلقائياً مع كل عملية)
  debit               Decimal             @default(0.00) @db.Decimal(10, 2)
  debitStatus         SupplierDebitStatus @default(PAID)
  paymentPeriodMonths Int                 @default(0)
  
  landlinePhone       String?
  mobilePhone         String
  
  settlementDate      DateTime?
  checksDueDate       DateTime?

  // العلاقات (Relations)
  transactions        SupplierTransaction[]
  payments            PaymentRecord[]
  // invoices         PurchaseInvoice[] // من الخطوات السابقة
  // returns          PurchaseReturn[]  // من الخطوات السابقة

  createdAt           DateTime            @default(now())
  updatedAt           DateTime            @updatedAt
}

// 2. سجل المعاملات "كشف الحساب" (يغني عن DebitLog)
model SupplierTransaction {
  id              String            @id @default(uuid())
  
  supplierId      String
  supplier        Supplier          @relation(fields: [supplierId], references: [id], onDelete: Restrict)
  supplierName    String            // Snapshot
  
  type            TransactionType
  status          TransactionStatus @default(COMPLETED)
  description     String
  
  amount          Decimal           @db.Decimal(10, 2)
  balanceBefore   Decimal           @db.Decimal(10, 2)
  balanceAfter    Decimal           @db.Decimal(10, 2)
  
  // الربط بالمستند الأصلي (فاتورة، إيصال دفع، أو مرتجع)
  // نستخدم String لأنه قد يشير إلى جداول مختلفة حسب نوع المعاملة
  referenceId     String?           
  referenceType   String?           // مثال: "PURCHASE", "PAYMENT"
  
  transactionDate DateTime          @default(now())
  notes           String?
  createdBy       String?           // معرف المستخدم الذي قام بالعملية

  createdAt       DateTime          @default(now())
  updatedAt       DateTime          @updatedAt
}

// 3. سجل المدفوعات (إيصالات الصرف للمورد)
model PaymentRecord {
  id              String        @id @default(uuid())
  
  supplierId      String
  supplier        Supplier      @relation(fields: [supplierId], references: [id], onDelete: Restrict)
  
  amount          Decimal       @db.Decimal(10, 2)
  paymentMethod   PaymentMethod
  paymentDate     DateTime
  dueDate         DateTime?     // تاريخ استحقاق الشيك
  
  checkNumber     String?
  bankName        String?
  notes           String?
  
  status          PaymentStatus @default(PAID)

  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt
}

// User Model
model User {
  id        String   @id @default(cuid())
  name      String?
  email     String   @unique
  password  String
  
  // Status: Active, Inactive, etc.
  isActive  Boolean  @default(true)
  
  // Timestamps
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // Relations
  roles     UserRole[]
}

// Role Model (e.g., Admin, Editor, Viewer)
model Role {
  id          String            @id @default(cuid())
  name        String            @unique
  description String?
  
  // Relations
  users       UserRole[]
  permissions RolePermission[]
}

// Permission Model (e.g., create_user, delete_user)
model Permission {
  id          String           @id @default(cuid())
  action      String           @unique // e.g., "USER_CREATE"
  description String?

  // Relations
  roles       RolePermission[]
}

// Junction Table for Many-to-Many (User <-> Role)
model UserRole {
  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId String
  role   Role   @relation(fields: [roleId], references: [id], onDelete: Cascade)
  roleId String

  @@id([userId, roleId])
}

// Junction Table for Many-to-Many (Role <-> Permission)
model RolePermission {
  role         Role       @relation(fields: [roleId], references: [id], onDelete: Cascade)
  roleId       String
  permission   Permission @relation(fields: [permissionId], references: [id], onDelete: Cascade)
  permissionId String

  @@id([roleId, permissionId])
}
